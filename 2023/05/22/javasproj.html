class Song {
  
  constructor(name, tID, rID, role="song") {
    this.name = name;
    this.tID = tID;
    this.rID = rID;
    this.role = role;
  }

  
  setRole(role) {
    this.role = role;
  }
  
  getJSON() {
    const obj = {type: typeof this, name: this.name, tID: this.tID, rID: this.rID, role: this.role};
    const json = JSON.stringify(obj);
    return json;
  }

 
  logIt() {

  }
    
}

    const lyrics = [];

    Papa.parse('files/lyrics.csv', {
      download: true,
      delimiter: ',',
      header: false,
      complete: function(results) {
        const rows = results.data;
        rows.forEach(row => {
          const [lyric, song] = row;
          lyrics.push({ lyric, song });
        });

        setupGame();
      },
      error: function(error) {
        console.log('Error:', error);
      }
    });

    function setupGame() {
      const randomIndex = Math.floor(Math.random() * lyrics.length);
      const selectedLyric = lyrics[randomIndex];

      const lyricElement = document.getElementById('lyric');
      lyricElement.textContent = selectedLyric.lyric;

      const submitButton = document.getElementById('submitBtn');
      const guessInput = document.getElementById('guessInput');
      submitButton.addEventListener('click', () => {
        const userGuess = guessInput.value;
        checkGuess(userGuess, selectedLyric.song);
      });

      function checkGuess(guess, correctAnswer) {
        if (guess.toLowerCase() === correctAnswer.toLowerCase()) {
          alert('Congratulations! You guessed correctly.');
          setupGame();
          guessInput.value = '';
        } else {
          alert('Oops! That was incorrect. Try again.');
        }
      }
    }
  </script>
